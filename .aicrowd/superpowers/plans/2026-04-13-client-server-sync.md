```python
#!/usr/bin/env python
"""Generate derived client artifacts from the core mechestim library.

Usage:
    uv run scripts/sync_client.py          # regenerate files in-place
    uv run scripts/sync_client.py --check  # exit non-zero if anything would change
"""
from __future__ import annotations

import argparse
import difflib
import sys
import textwrap
from pathlib import Path

_HEADER = '''\
# AUTO-GENERATED by scripts/sync_client.py — DO NOT EDIT
# Source of truth: the mechestim core library (src/mechestim/)
# Regenerate: uv run scripts/sync_client.py
'''

_CLIENT_SRC = Path(__file__).resolve().parent.parent / "mechestim-client" / "src" / "mechestim"


def _generate_registry_data() -> str:
    """Generate _registry_data.py from core REGISTRY."""
    from mechestim._registry import REGISTRY

    lines = [_HEADER, '"""Registry data snapshot for the mechestim client."""\n\n']
    lines.append("FUNCTION_CATEGORIES: dict[str, str] = {\n")
    for name in sorted(REGISTRY.keys()):
        cat = REGISTRY[name]["category"]
        lines.append(f'    {name!r}: {cat!r},\n')
    lines.append("}\n")
    return "".join(lines)


def _write_or_check(path: Path, content: str, check: bool, diffs: list[str]) -> None:
    """Write content to path, or check if it matches."""
    path.parent.mkdir(parents=True, exist_ok=True)
    if check:
        existing = path.read_text() if path.exists() else ""
        if existing != content:
            diff = difflib.unified_diff(
                existing.splitlines(keepends=True),
                content.splitlines(keepends=True),
                fromfile=str(path),
                tofile=str(path) + " (generated)",
            )
            diffs.append(f"--- {path} is out of sync ---\n" + "".join(diff))
    else:
        path.write_text(content)
        print(f"  wrote {path}")


def main():
    parser = argparse.ArgumentParser(description=__doc__)
    parser.add_argument("--check", action="store_true",
                        help="Exit non-zero if any file would change")
    args = parser.parse_args()
    diffs: list[str] = []

    print("Generating registry data...")
    _write_or_check(
        _CLIENT_SRC / "_registry_data.py",
        _generate_registry_data(),
        args.check, diffs,
    )

    if diffs:
        print("\n".join(diffs), file=sys.stderr)
        print(f"\n{len(diffs)} file(s) out of sync. Run: uv run scripts/sync_client.py",
              file=sys.stderr)
        sys.exit(1)
    elif args.check:
        print("All generated files are in sync.")


if __name__ == "__main__":
    main()
```

def _generate_errors() -> str:
    """Generate errors.py from core errors module."""
    import inspect
    import mechestim.errors as core_errors

    lines = [_HEADER, '"""Custom exceptions and warnings for the mechestim client."""\n\n']
    lines.append("from __future__ import annotations\n\n\n")

    # Collect all exception/warning classes defined in core errors
    classes = []
    for name, obj in inspect.getmembers(core_errors, inspect.isclass):
        if obj.__module__ == "mechestim.errors":
            classes.append((name, obj))

    # Sort by MRO depth (base classes first)
    classes.sort(key=lambda x: len(x[1].__mro__))

    for name, cls in classes:
        # Determine base class name
        base = cls.__bases__[0]
        if base.__module__ == "mechestim.errors":
            base_name = base.__name__
        elif base is Exception:
            base_name = "Exception"
        elif base is UserWarning:
            base_name = "UserWarning"
        else:
            base_name = base.__name__

        lines.append(f"class {name}({base_name}):\n")
        lines.append(f'    """{cls.__doc__}"""\n\n' if cls.__doc__ else '    pass\n\n')

        # If the class has a custom __init__, generate a message-based constructor
        # for the client (server sends error as a string message)
        if "__init__" in cls.__dict__:
            lines.append(f"    def __init__(self, message: str = \"\") -> None:\n")
            lines.append(f"        super().__init__(message)\n\n\n")
        else:
            lines.append("\n")

    # Add MechEstimServerError (client-only)
    lines.append("class MechEstimServerError(MechEstimError):\n")
    lines.append('    """Server-side error that does not map to a more specific exception."""\n\n\n')

    # Add error map and dispatcher
    lines.append("# ---------------------------------------------------------------------------\n")
    lines.append("# Error map and dispatcher\n")
    lines.append("# ---------------------------------------------------------------------------\n\n")

    mechestim_errors = [name for name, cls in classes
                        if issubclass(cls, core_errors.MechEstimError)
                        and name != "MechEstimError"]
    lines.append(f"_MECHESTIM_ERRORS = frozenset(\n    {{\n")
    for name in sorted(mechestim_errors):
        lines.append(f'        "{name}",\n')
    lines.append("    }\n)\n\n")

    lines.append("_ERROR_MAP: dict[str, type[Exception]] = {\n")
    for name in sorted(mechestim_errors):
        lines.append(f'    "{name}": {name},\n')
    lines.append('    "MechEstimServerError": MechEstimServerError,\n')
    lines.append('    "ValueError": ValueError,\n')
    lines.append('    "TypeError": TypeError,\n')
    lines.append('    "KeyError": KeyError,\n')
    lines.append('    "RuntimeError": RuntimeError,\n')
    lines.append('    "InvalidRequestError": MechEstimServerError,\n')
    lines.append("}\n\n\n")

    lines.append("def raise_from_response(error_type: str, message: str) -> None:\n")
    lines.append('    """Map *error_type* string to an exception class and raise it.\n\n')
    lines.append('    Unknown error types are treated as :class:`MechEstimServerError`.\n')
    lines.append('    """\n')
    lines.append("    if isinstance(message, bytes):\n")
    lines.append('        message = message.decode(\"utf-8\", errors=\"replace\")\n')
    lines.append("    if isinstance(error_type, bytes):\n")
    lines.append('        error_type = error_type.decode(\"utf-8\", errors=\"replace\")\n\n')
    lines.append("    exc_cls = _ERROR_MAP.get(error_type, MechEstimServerError)\n")
    lines.append("    if error_type in _MECHESTIM_ERRORS:\n")
    lines.append("        raise exc_cls(message=message)\n")
    lines.append("    raise exc_cls(message)\n")

    return "".join(lines)
```

def _generate_stats_init() -> str:
    """Generate stats/__init__.py with proxy distribution objects."""
    import mechestim.stats as core_stats

    dist_names = sorted(core_stats.__all__)

    lines = [_HEADER]
    lines.append('"""mechestim.stats — statistical distributions submodule proxy."""\n\n')
    lines.append("from __future__ import annotations\n\n")
    lines.append("from mechestim._connection import get_connection\n")
    lines.append("from mechestim._getattr import make_module_getattr\n")
    lines.append("from mechestim._protocol import encode_request\n")
    lines.append("from mechestim._remote_array import (\n")
    lines.append("    _encode_arg,\n")
    lines.append("    _result_from_response,\n")
    lines.append(")\n\n\n")

    # Generate distribution proxy class
    lines.append("class _DistributionProxy:\n")
    lines.append('    """Proxy for a server-side statistical distribution.\n\n')
    lines.append("    Provides ``.pdf()``, ``.cdf()``, ``.ppf()`` methods that\n")
    lines.append("    dispatch to the server.\n")
    lines.append('    """\n\n')
    lines.append("    def __init__(self, name: str) -> None:\n")
    lines.append("        self._name = name\n\n")
    lines.append("    def __repr__(self) -> str:\n")
    lines.append("        return f\"<mechestim.stats.{self._name}>\"\n\n")

    for method in ("pdf", "cdf", "ppf"):
        lines.append(f"    def {method}(self, x, *args, **kwargs):\n")
        lines.append(f'        """Dispatch stats.{{self._name}}.{method} to the server."""\n')
        lines.append(f"        conn = get_connection()\n")
        lines.append(f"        op = f\"stats.{{self._name}}.{method}\"\n")
        lines.append(f"        encoded_args = [_encode_arg(x)] + [_encode_arg(a) for a in args]\n")
        lines.append(f"        encoded_kwargs = {{k: _encode_arg(v) for k, v in kwargs.items()}}\n")
        lines.append(f"        resp = conn.send_recv(\n")
        lines.append(f"            encode_request(op, args=encoded_args, kwargs=encoded_kwargs)\n")
        lines.append(f"        )\n")
        lines.append(f"        return _result_from_response(resp)\n\n")

    # Generate instances
    lines.append("\n# Distribution instances\n")
    for name in dist_names:
        lines.append(f'{name} = _DistributionProxy({name!r})\n')

    lines.append(f"\n__all__ = {dist_names!r}\n\n")
    lines.append("# Fall-through for unknown names\n")
    lines.append('__getattr__ = make_module_getattr("stats.", "mechestim.stats")\n')

    return "".join(lines)
```

def _generate_linalg_init() -> str:
    """Generate linalg/__init__.py with proxies for all linalg ops."""
    from mechestim._registry import REGISTRY

    linalg_ops = sorted(
        name[len("linalg."):]
        for name, entry in REGISTRY.items()
        if name.startswith("linalg.") and entry["category"] != "blacklisted"
    )

    lines = [_HEADER]
    lines.append('"""mechestim.linalg — linear algebra submodule proxy."""\n\n')
    lines.append("from __future__ import annotations\n\n")
    lines.append("from mechestim._connection import get_connection\n")
    lines.append("from mechestim._getattr import make_module_getattr\n")
    lines.append("from mechestim._protocol import encode_request\n")
    lines.append("from mechestim._remote_array import (\n")
    lines.append("    _encode_arg,\n")
    lines.append("    _result_from_response,\n")
    lines.append(")\n\n\n")

    lines.append("def _make_linalg_proxy(op_name: str):\n")
    lines.append('    """Create a proxy function for ``linalg.<op_name>``."""\n')
    lines.append('    qualified = f"linalg.{op_name}"\n\n')
    lines.append("    def proxy(*args, **kwargs):\n")
    lines.append("        conn = get_connection()\n")
    lines.append("        encoded_args = [_encode_arg(a) for a in args]\n")
    lines.append("        encoded_kwargs = {k: _encode_arg(v) for k, v in kwargs.items()}\n")
    lines.append("        resp = conn.send_recv(\n")
    lines.append("            encode_request(qualified, args=encoded_args, kwargs=encoded_kwargs)\n")
    lines.append("        )\n")
    lines.append("        return _result_from_response(resp)\n\n")
    lines.append('    proxy.__name__ = op_name\n')
    lines.append('    proxy.__qualname__ = f"linalg.{op_name}"\n')
    lines.append("    return proxy\n\n\n")

    for op in linalg_ops:
        lines.append(f'{op} = _make_linalg_proxy({op!r})\n')

    lines.append('\n# Fall-through for blacklisted / unknown names\n')
    lines.append('__getattr__ = make_module_getattr("linalg.", "mechestim.linalg")\n')

    return "".join(lines)
```

"""Pure-Python SymmetryInfo for the mechestim client.

This is a read-only metadata container that records symmetry information
reported by the server. It does NOT include SymmetricTensor or the
computed properties that require numpy.
"""
from __future__ import annotations

from dataclasses import dataclass, field


@dataclass(frozen=True)
class SymmetryInfo:
    """Metadata about tensor symmetry groups.

    Parameters
    ----------
    symmetric_axes : list of tuple of int
        Groups of dimension indices that are symmetric under permutation.
    shape : tuple of int
        Full tensor shape.
    """

    symmetric_axes: list[tuple[int, ...]] = field(default_factory=list)
    shape: tuple[int, ...] = ()

    def __post_init__(self) -> None:
        normalized = [tuple(sorted(g)) for g in self.symmetric_axes]
        object.__setattr__(self, "symmetric_axes", normalized)
```
