```python
#!/usr/bin/env python3
"""Migrate MkDocs markdown content to Docusaurus format.

Usage:
    python scripts/migrate_to_docusaurus.py          # run migration
    python scripts/migrate_to_docusaurus.py --dry-run # preview changes
"""

import argparse
import json
import re
import shutil
from pathlib import Path

ROOT = Path(__file__).resolve().parent.parent
MKDOCS_DOCS = ROOT / "docs"
WEBSITE_DOCS = ROOT / "website" / "docs"

# Navigation order from mkdocs.yml — maps directory/file to sidebar_position
NAV_ORDER: dict[str, list[dict[str, str]]] = {
    "getting-started": [
        {"installation.md": "Installation"},
        {"first-budget.md": "Your First Budget"},
    ],
    "how-to": [
        {"migrate-from-numpy.md": "Migrate from NumPy"},
        {"use-einsum.md": "Use Einsum"},
        {"exploit-symmetry.md": "Exploit Symmetry"},
        {"use-linalg.md": "Use Linear Algebra"},
        {"use-fft.md": "Use FFT"},
        {"plan-your-budget.md": "Plan Your Budget"},
        {"calibrate-weights.md": "Calibrate Weights"},
        {"debug-budget-overruns.md": "Debug Budget Overruns"},
    ],
    "concepts": [
        {"flop-counting-model.md": "FLOP Counting Model"},
        {"operation-categories.md": "Operation Categories"},
        {"numpy-compatibility-testing.md": "NumPy Compatibility Testing"},
    ],
    "architecture": [
        {"client-server.md": "Client-Server Model"},
        {"docker.md": "Running with Docker"},
    ],
    "explanation": [
        {"subgraph-symmetry.md": "Subgraph Symmetry Detection"},
        {"symmetry-explorer.md": "Symmetry Explorer"},
    ],
    "development": [
        {"contributing.md": "Contributor Guide"},
    ],
    "reference": [
        {"for-agents.md": "For AI Agents"},
        {"operation-audit.md": "Operation Audit"},
        {"empirical-weights.md": "FLOP Weight Calibration Results"},
        {"cheat-sheet.md": "FLOP Cost Cheat Sheet"},
    ],
    "troubleshooting": [
        {"common-errors.md": "Common Errors"},
    ],
}

# Category labels and positions for _category_.json
CATEGORY_META: dict[str, tuple[int, str]] = {
    "getting-started": (1, "Getting Started"),
    "how-to": (2, "How-To Guides"),
    "concepts": (3, "Concepts"),
    "architecture": (4, "Architecture"),
    "explanation": (5, "Explanation"),
    "development": (6, "Development"),
    "api": (7, "API Reference"),
    "reference": (8, "Reference"),
    "troubleshooting": (9, "Troubleshooting"),
}

# Files to skip (handled separately or deleted)
SKIP_FILES = {
    "index.md",  # replaced by custom landing page
}

SKIP_DIRS = {
    "visualization",  # symmetry explorer absorbed into website/src/
    "javascripts",    # replaced by Docusaurus plugins
    "stylesheets",    # ported to website/src/css/
    "assets",         # logo copied separately
}


def convert_admonitions(content: str) -> str:
    """Convert MkDocs admonitions to Docusaurus format.

    !!! type "title"        →  :::type[title]
        content                content
                               :::
    """
    lines = content.split("\n")
    result = []
    i = 0
    while i < len(lines):
        # Match admonition opener: !!! type "title" or !!! type
        m = re.match(r'^(!{3})\s+(\w+)\s*"([^"]*)"', lines[i])
        if not m:
            m = re.match(r'^(!{3})\s+(\w+)\s*$', lines[i])
        if m:
            admon_type = m.group(2)
            title = m.group(3) if len(m.groups()) >= 3 else ""
            if title:
                result.append(f":::{admon_type}[{title}]")
            else:
                result.append(f":::{admon_type}")
            i += 1
            # Collect indented content (4 spaces)
            while i < len(lines) and (lines[i].startswith("    ") or lines[i].strip() == ""):
                if lines[i].strip() == "":
                    result.append("")
                else:
                    result.append(lines[i][4:])  # remove 4-space indent
                i += 1
            result.append(":::")
        else:
            result.append(lines[i])
            i += 1
    return "\n".join(result)


def add_frontmatter(content: str, sidebar_position: int, sidebar_label: str) -> str:
    """Add Docusaurus frontmatter to a markdown file."""
    # Check if file already has frontmatter
    if content.startswith("---"):
        return content

    frontmatter = f"""---
sidebar_position: {sidebar_position}
sidebar_label: "{sidebar_label}"
---

"""
    return frontmatter + content


def get_sidebar_info(directory: str, filename: str) -> tuple[int, str] | None:
    """Look up sidebar position and label from NAV_ORDER."""
    entries = NAV_ORDER.get(directory, [])
    for i, entry in enumerate(entries):
        for fname, label in entry.items():
            if fname == filename:
                return (i + 1, label)
    return None


def migrate_file(src: Path, dst: Path, dry_run: bool = False) -> None:
    """Migrate a single markdown file."""
    content = src.read_text()

    # Convert admonitions
    content = convert_admonitions(content)

    # Determine sidebar info from path
    rel = src.relative_to(MKDOCS_DOCS)
    parts = rel.parts
    if len(parts) >= 2:
        directory = parts[0]
        filename = parts[-1]
        info = get_sidebar_info(directory, filename)
        if info:
            position, label = info
            content = add_frontmatter(content, position, label)

    if dry_run:
        print(f"  Would write: {dst}")
        return

    dst.parent.mkdir(parents=True, exist_ok=True)
    dst.write_text(content)


def generate_category_json(dry_run: bool = False) -> None:
    """Generate _category_.json files for each docs subdirectory."""
    for dirname, (position, label) in CATEGORY_META.items():
        cat_dir = WEBSITE_DOCS / dirname
        cat_file = cat_dir / "_category_.json"
        data = {
            "label": label,
            "position": position,
            "collapsed": False,
        }
        if dry_run:
            print(f"  Would write: {cat_file}")
            continue
        cat_dir.mkdir(parents=True, exist_ok=True)
        cat_file.write_text(json.dumps(data, indent=2) + "\n")


def main():
    parser = argparse.ArgumentParser(description="Migrate MkDocs docs to Docusaurus")
    parser.add_argument("--dry-run", action="store_true", help="Preview without writing")
    args = parser.parse_args()

    print("Migrating MkDocs content to Docusaurus...")

    # Generate _category_.json files
    print("\nGenerating _category_.json files:")
    generate_category_json(dry_run=args.dry_run)

    # Migrate markdown files
    print("\nMigrating markdown files:")
    migrated = 0
    skipped = 0
    for src in sorted(MKDOCS_DOCS.rglob("*.md")):
        rel = src.relative_to(MKDOCS_DOCS)

        # Skip files in excluded directories
        if any(part in SKIP_DIRS for part in rel.parts):
            print(f"  SKIP (excluded dir): {rel}")
            skipped += 1
            continue

        # Skip specific files
        if rel.name in SKIP_FILES or str(rel) in SKIP_FILES:
            print(f"  SKIP (excluded file): {rel}")
            skipped += 1
            continue

        dst = WEBSITE_DOCS / rel
        migrate_file(src, dst, dry_run=args.dry_run)
        print(f"  {'Would migrate' if args.dry_run else 'Migrated'}: {rel}")
        migrated += 1

    # Copy changelog to root
    changelog_src = MKDOCS_DOCS / "changelog.md"
    if changelog_src.exists():
        content = changelog_src.read_text()
        content = add_frontmatter(content, 99, "Changelog")
        if not args.dry_run:
            (WEBSITE_DOCS / "changelog.md").write_text(content)
        print(f"  {'Would migrate' if args.dry_run else 'Migrated'}: changelog.md")

    print(f"\nDone: {migrated} migrated, {skipped} skipped")


if __name__ == "__main__":
    main()
```

Create `website/src/components/shared/SortableTable.tsx`:

```tsx
import React, {useState, useMemo, useCallback} from 'react';
import styles from './SortableTable.module.css';

type SortDirection = 'asc' | 'desc' | null;

interface Column {
  key: string;
  label: string;
  /** If true, sort numerically (strips commas, parses as float) */
  numeric?: boolean;
}

interface SortableTableProps {
  columns: Column[];
  data: Record<string, string | number>[];
  /** Show a search/filter input above the table */
  filterable?: boolean;
  /** Placeholder text for the filter input */
  filterPlaceholder?: string;
  /** Make the header sticky on scroll */
  stickyHeader?: boolean;
}

function parseNumeric(val: string | number): number {
  if (typeof val === 'number') return val;
  const cleaned = val.replace(/,/g, '').replace(/[^\d.\-eE+]/g, '');
  const n = parseFloat(cleaned);
  return isNaN(n) ? 0 : n;
}

export default function SortableTable({
  columns,
  data,
  filterable = false,
  filterPlaceholder = 'Filter...',
  stickyHeader = false,
}: SortableTableProps) {
  const [sortKey, setSortKey] = useState<string | null>(null);
  const [sortDir, setSortDir] = useState<SortDirection>(null);
  const [filter, setFilter] = useState('');

  const handleSort = useCallback((key: string) => {
    if (sortKey === key) {
      setSortDir(prev => prev === 'asc' ? 'desc' : prev === 'desc' ? null : 'asc');
      if (sortDir === 'desc') setSortKey(null);
    } else {
      setSortKey(key);
      setSortDir('asc');
    }
  }, [sortKey, sortDir]);

  const filtered = useMemo(() => {
    if (!filter) return data;
    const lower = filter.toLowerCase();
    return data.filter(row =>
      columns.some(col => String(row[col.key]).toLowerCase().includes(lower))
    );
  }, [data, filter, columns]);

  const sorted = useMemo(() => {
    if (!sortKey || !sortDir) return filtered;
    const col = columns.find(c => c.key === sortKey);
    return [...filtered].sort((a, b) => {
      const aVal = a[sortKey];
      const bVal = b[sortKey];
      let cmp: number;
      if (col?.numeric) {
        cmp = parseNumeric(aVal) - parseNumeric(bVal);
      } else {
        cmp = String(aVal).localeCompare(String(bVal));
      }
      return sortDir === 'desc' ? -cmp : cmp;
    });
  }, [filtered, sortKey, sortDir, columns]);

  const sortIndicator = (key: string) => {
    if (sortKey !== key) return ' ↕';
    return sortDir === 'asc' ? ' ↑' : ' ↓';
  };

  return (
    <div className={styles.wrapper}>
      {filterable && (
        <input
          className={styles.filterInput}
          type="text"
          placeholder={filterPlaceholder}
          value={filter}
          onChange={e => setFilter(e.target.value)}
        />
      )}
      <div className={styles.tableContainer}>
        <table className={styles.table}>
          <thead className={stickyHeader ? styles.stickyHead : undefined}>
            <tr>
              {columns.map(col => (
                <th
                  key={col.key}
                  className={styles.th}
                  onClick={() => handleSort(col.key)}
                >
                  {col.label}{sortIndicator(col.key)}
                </th>
              ))}
            </tr>
          </thead>
          <tbody>
            {sorted.map((row, i) => (
              <tr key={i}>
                {columns.map(col => (
                  <td key={col.key} className={styles.td}>{row[col.key]}</td>
                ))}
              </tr>
            ))}
          </tbody>
        </table>
      </div>
      <div className={styles.count}>{sorted.length} of {data.length} rows</div>
    </div>
  );
}
```

- [ ] **Step 7: Create the main ApiReference component**

Create `website/src/components/api-reference/ApiReference.tsx`:

```tsx
import React, {useState, useMemo} from 'react';
import {usePluginData} from '@docusaurus/useGlobalData';
import FilterBar from './FilterBar';
import OperationRow from './OperationRow';

interface Operation {
  name: string;
  module: string;
  whest_ref: string;
  numpy_ref: string;
  category: string;
  cost_formula: string;
  cost_formula_latex: string;
  free: boolean;
  blocked: boolean;
  status: string;
  notes: string;
}

interface OpsData {
  operations: Operation[];
}

export default function ApiReference() {
  const data = usePluginData('plugin-api-docs') as OpsData;
  const operations = data.operations;

  const [search, setSearch] = useState('');
  const [filter, setFilter] = useState<'all' | 'counted' | 'free' | 'blocked'>('all');
  const [moduleFilter, setModuleFilter] = useState('');

  const modules = useMemo(() => {
    const set = new Set(operations.map(op => op.module));
    return [...set].sort();
  }, [operations]);

  const filtered = useMemo(() => {
    return operations.filter(op => {
      if (search) {
        const lower = search.toLowerCase();
        if (!op.name.toLowerCase().includes(lower) &&
            !op.whest_ref.toLowerCase().includes(lower) &&
            !op.notes.toLowerCase().includes(lower)) {
          return false;
        }
      }
      if (filter === 'counted' && (op.free || op.blocked)) return false;
      if (filter === 'free' && !op.free) return false;
      if (filter === 'blocked' && !op.blocked) return false;
      if (moduleFilter && op.module !== moduleFilter) return false;
      return true;
    });
  }, [operations, search, filter, moduleFilter]);

  return (
    <div>
      <FilterBar
        search={search}
        onSearchChange={setSearch}
        filter={filter}
        onFilterChange={setFilter}
        moduleFilter={moduleFilter}
        onModuleChange={setModuleFilter}
        modules={modules}
        resultCount={filtered.length}
        totalCount={operations.length}
      />
      <div style={{overflowX: 'auto'}}>
        <table style={{width: '100%', borderCollapse: 'collapse', fontSize: '0.88rem'}}>
          <thead>
            <tr style={{background: '#f8f9fa'}}>
              <th style={{padding: '10px 12px', textAlign: 'left', fontWeight: 600, borderBottom: '2px solid #e5e7eb'}}>Operation</th>
              <th style={{padding: '10px 12px', textAlign: 'left', fontWeight: 600, borderBottom: '2px solid #e5e7eb'}}>Module</th>
              <th style={{padding: '10px 12px', textAlign: 'left', fontWeight: 600, borderBottom: '2px solid #e5e7eb'}}>Type</th>
              <th style={{padding: '10px 12px', textAlign: 'left', fontWeight: 600, borderBottom: '2px solid #e5e7eb'}}>Cost Formula</th>
              <th style={{padding: '10px 12px', textAlign: 'center', fontWeight: 600, borderBottom: '2px solid #e5e7eb', width: 40}}></th>
            </tr>
          </thead>
          <tbody>
            {filtered.map(op => (
              <OperationRow key={op.whest_ref} op={op} />
            ))}
          </tbody>
        </table>
      </div>
    </div>
  );
}
```

function extractText(html) {
  const dom = new JSDOM(html);
  const article = dom.window.document.querySelector('article') ||
                  dom.window.document.querySelector('main') ||
                  dom.window.document.body;
  // Remove script and style tags
  article.querySelectorAll('script, style, nav').forEach(el => el.remove());
  return article.textContent.replace(/\s+/g, ' ').trim();
}

module.exports = function pluginLlmsTxt(context, options) {
  const sections = options.sections || DEFAULT_SECTIONS;

  return {
    name: 'plugin-llmstxt',

    async postBuild({outDir}) {
      const baseUrl = context.siteConfig.baseUrl.replace(/\/$/, '');

      // Generate llms.txt (index)
      let llmsTxt = `# ${context.siteConfig.title}\n\n`;
      llmsTxt += `> ${context.siteConfig.tagline}\n\n`;

      for (const [sectionName, pages] of Object.entries(sections)) {
        llmsTxt += `## ${sectionName}\n\n`;
        for (const [slug, description] of Object.entries(pages)) {
          const url = `${context.siteConfig.url}${baseUrl}/${slug}`;
          llmsTxt += `- [${description}](${url})\n`;
        }
        llmsTxt += '\n';
      }

      fs.writeFileSync(path.join(outDir, 'llms.txt'), llmsTxt);

      // Generate llms-full.txt (all content)
      let llmsFullTxt = `# ${context.siteConfig.title}\n\n`;
      llmsFullTxt += `> ${context.siteConfig.tagline}\n\n`;

      for (const [sectionName, pages] of Object.entries(sections)) {
        llmsFullTxt += `## ${sectionName}\n\n`;
        for (const [slug, description] of Object.entries(pages)) {
          // Try to find the built HTML file
          const possiblePaths = [
            path.join(outDir, slug, 'index.html'),
            path.join(outDir, slug + '.html'),
          ];
          let htmlContent = null;
          for (const p of possiblePaths) {
            if (fs.existsSync(p)) {
              htmlContent = fs.readFileSync(p, 'utf8');
              break;
            }
          }
          if (htmlContent) {
            const text = extractText(htmlContent);
            llmsFullTxt += `### ${description}\n\n${text}\n\n`;
          } else {
            llmsFullTxt += `### ${description}\n\n(Content not available)\n\n`;
          }
        }
      }

      fs.writeFileSync(path.join(outDir, 'llms-full.txt'), llmsFullTxt);

      console.log('[plugin-llmstxt] Generated llms.txt and llms-full.txt');
    },
  };
};
